## Build Chat App with React, Socket.io, Nodejs, Redux-Toolkit, MongoDB (2024)

# Server .env file

FRONTEND_URL = <Frontend URL>

# client .env file

# Quick start

# to run server

cd server

npm i

npm start

# to run client

cd client

npm i

npm start
